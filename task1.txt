olegkrilov@olegkrilov-VirtualBox:~/les6/task1$ nano some-script.sh
olegkrilov@olegkrilov-VirtualBox:~/les6/task1$ cat some-script.sh
#!/bin/bash
echo 'Initial content:'
cat some-text-file
echo '---------------------------------------'
echo 'First: replace all chars with uppercase'
tr '[:lower:]' '[:upper:]' < some-text-file > some-text-file-UPPERCASE
cat some-text-file-UPPERCASE
echo '---------------------------------------'
echo 'Now replace "newline"symbols'
tr "\n" "\t" < some-text-file-UPPERCASE > some-text-file-ONELINE
cat some-text-file-ONELINE
