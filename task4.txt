olegkrilov@olegkrilov-VirtualBox:~/les6/task4$ touch syslog-term-analyzer.sh
olegkrilov@olegkrilov-VirtualBox:~/les6/task4$ chmod +x syslog-term-analyzer.sh
olegkrilov@olegkrilov-VirtualBox:~/les6/task4$ nano syslog-term-analyzer.sh
olegkrilov@olegkrilov-VirtualBox:~/les6/task4$ cat syslog-term-analyzer.sh
#!/bin/bash
SEARCH_KEY=${1:-SIGRTMIN}

echo 'Observe syslogs filtered by' $SEARCH_KEY
cat /var/log/syslog | grep $SEARCH_KEY
